```typescript
// Player.ts - 点击移动
onTouchMove(event: EventTouch) {
    const targetPos = event.getUILocation();
    this.targetPosition = targetPos;
    this.moveToTarget();
}

// Player.ts - 自动瞄准射击
findNearestEnemy(): Enemy | null {
    // 查找射程内最近敌人
}

autoShoot() {
    const target = this.findNearestEnemy();
    if (target) this.fire(target);
}
```
